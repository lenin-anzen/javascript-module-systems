CommonJS

  • Es un formato donde cada módulo es ejecutado en su propio namespace.
  • Es síncrono

ECMAScript 5

// es5/module.js module.exports = function( value ){ return value*2; }
// es5/main.js var multiplyBy2 = require('./module'); var result = multiplyBy2( 4 ); console.log(result);

ECMAScript 6

// es6/module.js export default function( value ){ return value*2; }
// es6/main.js import multiplyBy2 from './module'; const result = multiplyBy2( 4 ); console.log( result );
