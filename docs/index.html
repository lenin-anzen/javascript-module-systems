CommonJS

  • Es un formato donde cada módulo es ejecutado en su propio namespace.
  • Es síncrono

ECMAScript 5

// src/commonjs/es5/module.js module.exports = function( value ){ return value*2; }
// src/commonjs/es5/main.js var multiplyBy2 = require('./module'); var result = multiplyBy2( 4 ); console.log('ECMAScript 5 / CommonJS:', result);

ECMAScript 6

// src/commonjs/es6/module.js export default function( value ){ return value*2; }
// src/commonjs/es6/main.js import multiplyBy2 from './module'; const result = multiplyBy2( 4 ); console.log('ECMAScript 6 / CommonJS:', result);

AMD: Asynchronous Module Definition

  • Es un fork de CommonJS
  • Es Asíncrono

Client side:

Instalación de RequireJS:

Agregar referencia del script al HTML.

<script data-main="src/amd/es5/main" src="https://cdnjs.cloudflare.com/ajax/libs/require.js/2.3.3/require.min.js"></script>

Crear el archivo inicial que cargará RequireJS

// src/amd/es5/main.js requirejs(["multiplyBy2", "sumFour"], function(multiplyBy2, sumFour) { var result1 = multiplyBy2( 4 ); console.log( 'ECMAScript 5 / AMD:', result1 ); var result2 = sumFour( 3 ); console.log( 'ECMAScript 5 / AMD:', result2 ); });

Crear los archivos que cargará main.js

// src/amd/es5/multiplyBy2.js define(function (require, exports, module) { module.exports = function( value ){ return value*2; } });
// src/amd/es5/sumFour.js define(function() { return function( value ) { return value + 4; } });
